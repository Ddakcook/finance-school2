    const GAS_WEB_APP_URL = 'https://script.google.com/macros/s/AKfycbwZmr5C_v-no6eIVErRnR-235dZPU6HJvVeiPHh64o8LAbSpKSqZM_fkyuVbgcuuThk/exec';

    function onlyNumber(value) {

      return Number(String(value).replace(/[^0-9.-]/g, '')) || 0;

    }


    function money(value) {

      return Number(value || 0).toLocaleString('ko-KR') + '원';

    }


    function convertToKoreanCurrency(num) {

      if (!num || isNaN(num)) return '';

      if (num >= 100000000) {

        const eok = Math.floor(num / 100000000);

        const man = Math.floor((num % 100000000) / 10000);

        return man > 0 ? `${eok}억 ${man.toLocaleString('ko-KR')}만 원` : `${eok}억 원`;

      } else if (num >= 10000) {

        const man = Math.floor(num / 10000);

        return `${man.toLocaleString('ko-KR')}만 원`;

      }

      return num.toLocaleString('ko-KR') + '원';

    }


    function handleMoneyInput(input, guideId) {

      const raw = String(input.value).replace(/[^0-9]/g, '');

      input.value = raw ? Number(raw).toLocaleString('ko-KR') : '';

      const numValue = Number(raw);

      document.getElementById(guideId).innerText = numValue ? `익는 중: ${convertToKoreanCurrency(numValue)}` : '';

    }


    function setStatus(text) {

      document.getElementById('status').innerText = text;

    }


    function normalizeRate(sign, value) {

      const clean = String(value || '').replace(/[+-%\s]/g, '');

      if (!clean) return '';

      return sign + clean + '%';

    }


    function splitRate(rate) {

      const text = String(rate || '').trim();

      return {

        sign: text.startsWith('-') ? '-' : '+',

        value: text.replace(/[+-%\s]/g, '')

      };

    }


    function addStockItem(data) {

      data = data || {};

      const rate = splitRate(data.rate || '');


      const box = document.createElement('div');

      box.className = 'stock-item';


      const name = document.createElement('input');

      name.className = 'stock-name';

      name.placeholder = '종목명';

      name.value = data.name || '';


      const price = document.createElement('input');

      price.className = 'stock-price';

      price.placeholder = '매수가';

      price.value = data.price ? Number(data.price).toLocaleString('ko-KR') : '';

      price.oninput = function() {

        const raw = String(price.value).replace(/[^0-9]/g, '');

        price.value = raw ? Number(raw).toLocaleString('ko-KR') : '';

      };


      const sign = document.createElement('select');

      sign.className = 'stock-rate-sign';

      sign.innerHTML = '<option value="+">+</option><option value="-">-</option>';

      sign.value = rate.sign;


      const rateInput = document.createElement('input');

      rateInput.className = 'stock-rate';

      rateInput.placeholder = '수익률';

      rateInput.value = rate.value;


      const round = document.createElement('select');

      round.className = 'stock-round';

      round.innerHTML =

        '<option value="1차">1차</option>' +

        '<option value="2차">2차</option>' +

        '<option value="3차">3차</option>' +

        '<option value="4차">4차</option>';

      round.value = data.round || '1차';


      const del = document.createElement('button');

      del.type = 'button';

      del.className = 'delete-btn';

      del.innerText = '×';

      del.onclick = function() { box.remove(); };


      box.appendChild(name);

      box.appendChild(price);

      box.appendChild(sign);

      box.appendChild(rateInput);

      box.appendChild(del);


      document.getElementById('stockList').appendChild(box);

    }

    function loadData() {

      setStatus('DB 불러오는 중');


      fetch(GAS_WEB_APP_URL)

        .then(res => res.json())

        .then(data => {

          renderTable();

          setStatus('DB 연결 정상');

        })

        .catch(error => {

          setStatus('불러오기 실패');

          alert('구글 시트 연동 실패: ' + error.message + '\n구글 웹앱 배포 및 URL 주소를 다시 확인해 주세요.');

        });

    }


    function loadPersonByName() {

      if (currentId) return; 

      const name = document.getElementById('name').value.trim();

      if (!name) return;


      const person = people.find(p => p.name === name);

      if (person) fillForm(person);

    }


    function fillForm(p) {

      currentId = p.id || ''; 

      document.getElementById('name').value = p.name || '';

      document.getElementById('money').value = p.totalMoney ? Number(p.totalMoney).toLocaleString('ko-KR') : '';

      document.getElementById('cash').value = p.cash ? Number(p.cash).toLocaleString('ko-KR') : '';

      

      document.getElementById('moneyGuide').innerText = p.totalMoney ? `익는 중: ${convertToKoreanCurrency(p.totalMoney)}` : '';

      document.getElementById('cashGuide').innerText = p.cash ? `익는 중: ${convertToKoreanCurrency(p.cash)}` : '';

      

      document.getElementById('memo').value = p.memo || '';

      document.getElementById('stockList').innerHTML = '';

      document.getElementById('waitList').innerHTML = '';


      if (p.stocks && p.stocks.length) p.stocks.forEach(addStockItem);

      else addStockItem();


      if (p.waits && p.waits.length) p.waits.forEach(addWaitItem);

      else addWaitItem();


      document.getElementById('editMode').style.display = 'block';

      window.scrollTo({ top: 0, behavior: 'smooth' });

    }


    function collectStocks() {

      return Array.from(document.querySelectorAll('.stock-item'))

        .map(row => ({

          name: row.querySelector('.stock-name').value.trim(),

          price: onlyNumber(row.querySelector('.stock-price').value),

          rate: normalizeRate(

            row.querySelector('.stock-rate-sign').value,

            row.querySelector('.stock-rate').value

          ),

          round: row.querySelector('.stock-round').value

        }))

        .filter(x => x.name); 

    }


    function collectWaits() {

      return Array.from(document.querySelectorAll('.wait-name'))

        .map(x => x.value.trim())

        .filter(Boolean);

    }


    function saveData() {

      const name = document.getElementById('name').value.trim();

      if (!name) { alert('이름을 입력해주세요.'); return; }


      const person = {

        id: currentId, 

        name: name,

        totalMoney: onlyNumber(document.getElementById('money').value),

        cash: onlyNumber(document.getElementById('cash').value),

        memo: document.getElementById('memo').value.trim() || '변동 0',

        stocks: collectStocks(),

        waits: collectWaits()

      };


      document.getElementById('saveBtn').disabled = true;

      document.getElementById('saveBtn').innerText = '저장 중...';

      setStatus('저장 중');


      fetch(GAS_WEB_APP_URL, {

        method: 'POST',

        body: JSON.stringify({ action: 'save', person: person })

      })

      .then(res => res.json())

      .then(res => {

        document.getElementById('saveBtn').disabled = false;

        document.getElementById('saveBtn').innerText = '저장하기';

        if(res.success) {

          setStatus('저장 완료');

          resetForm();

          loadData();

        } else { alert('저장 실패: ' + res.error); }

      })

      .catch(error => {

        document.getElementById('saveBtn').disabled = false;

        document.getElementById('saveBtn').innerText = '저장하기';

        setStatus('저장 실패');

        alert('통신 에러가 발생했습니다.\n구글 웹앱 권한 설정을 "Anyone"으로 하셨는지 확인해 주세요.');

      });

    }


    function renderTable() {

      const tbody = document.getElementById('tbody');

      tbody.innerHTML = '';


      const sorted = people.slice().sort((a, b) => a.name.localeCompare(b.name, 'ko'));


      if (!sorted.length) {

        tbody.innerHTML = '<tr><td colspan="6" class="empty">저장된 데이터가 없습니다.</td></tr>';

        return;

      }


      sorted.forEach(function(p) {

        const stockHtml = (p.stocks || []).map(function(s) {

          const cls = String(s.rate || '').startsWith('-') ? 'minus' : 'plus';

          return '<span class="badge ' + cls + '">' +

            s.name + ' ' + money(s.price) + ' ' + (s.rate || '') + ' ' + (s.round || '') +

            '</span>';

        }).join('');


        const waitHtml = (p.waits || []).map(function(w) {

          return '<span class="badge wait">' + w + '</span>';

        }).join('');


        const tr = document.createElement('tr');

        

        const nameTd = document.createElement('td');

        nameTd.className = 'name-cell';

        nameTd.textContent = p.name; 


        tr.appendChild(nameTd);

        

        const restContents = 

          '<td>' + money(p.totalMoney) + '</td>' +

          '<td>' + money(p.cash) + '</td>' +

          '<td>' + (stockHtml || '-') + '</td>' +

          '<td>' + (waitHtml || '-') + '</td>' +

          '<td><div class="action-cell">' +

          '<button type="button" class="btn-sub btn-small src-edit-btn">수정</button>' +

          '<button type="button" class="btn-danger btn-small src-del-btn">삭제</button>' +

          '</div></td>';

          

        tr.insertAdjacentHTML('beforeend', restContents);


        tr.querySelector('.src-edit-btn').onclick = function() { fillForm(p); };

        tr.querySelector('.src-del-btn').onclick = function() { deleteData(p.id); };


        tbody.appendChild(tr);

      });

    }


    function deleteData(id) {

      if (!confirm('삭제하시겠습니까?')) return;

      setStatus('삭제 중');


      fetch(GAS_WEB_APP_URL, {

        method: 'POST',

        body: JSON.stringify({ action: 'delete', id: id })

      })

      .then(res => res.json())

      .then(res => {

        if(res.success) {

          setStatus('삭제 완료');

          loadData();

        } else { alert('삭제 실패: ' + res.error); }

      });

    }


    function clearAll() {

      if (!confirm('정말 전체 데이터를 삭제하시겠습니까?')) return;

      setStatus('전체 삭제 중');


      fetch(GAS_WEB_APP_URL, {

        method: 'POST',

        body: JSON.stringify({ action: 'clear' })

      })

      .then(res => res.json())

      .then(res => {

        if(res.success) {

          setStatus('전체 삭제 완료');

          resetForm();

          loadData();

        } else { alert('삭제 실패: ' + res.error); }

      });

    }


    window.onload = function() {

      resetForm();

      loadData();

    };
